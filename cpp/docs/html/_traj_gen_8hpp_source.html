363  template<typename T>
364  VectorX<T> QpBlock<T>::getQpSol(bool & isSolved) {
365  uint nVar = Q.rows();
366  int N_ineq_const = A.rows();
367  int N_eq_const = Aeq.rows();
368  int N_const = N_ineq_const + N_eq_const;
369  // convention follows the qpOASES
370  qpOASES::QProblem qp_obj(nVar, N_const, qpOASES::HST_SEMIDEF);
371  Matrix<double, -1, -1, RowMajor> Hmat(2 * Q.template cast<double>());
372  Matrix<double, 1, -1, RowMajor> gmat(H.template cast<double>());
373  Matrix<double, -1, -1, RowMajor> Amat(N_const, nVar);
374  Amat << MatrixXd(A.template cast<double>()), MatrixXd(Aeq.template cast<double>());
375  Matrix<double, -1, -1, RowMajor> ubAmat(N_const, 1);
376  ubAmat << MatrixXd(b.template cast<double>()), MatrixXd(beq.template cast<double>());
377  MatrixXd MinusInf(N_ineq_const, 1);
378  MinusInf=MinusInf.setConstant(-99999);
379  Matrix<double, -1, -1, RowMajor> lbAmat(N_const, 1);
380  lbAmat << MinusInf, MatrixXd(beq.template cast<double>());
381 
382  qpOASES::real_t *Hqp = Hmat.data();
383  qpOASES::real_t *gqp = gmat.data();
384  qpOASES::real_t *Aqp = Amat.data();
385  qpOASES::real_t *lbAqp = lbAmat.data();
386  qpOASES::real_t *ubAqp = ubAmat.data();
387 
388  qpOASES::Options options;
389  options.printLevel = qpOASES::PL_LOW;
390  options.terminationTolerance = 1e-9;
391  options.enableRamping=qpOASES::BT_FALSE;
392  qp_obj.setOptions(options);
393  qpOASES::int_t nWSR = 2000;
394  qp_obj.init(Hqp, gqp, Aqp, NULL, NULL, lbAqp, ubAqp, nWSR);
395  if(qp_obj.isInfeasible()){
396  cout<<"[QP solver] warning: problem is infeasible. "<<endl;
397  }
398 
399  cout << "Acutual nWSR: " << nWSR << " / Original: " << 2000 << endl;
400 
401  qpOASES::real_t xOpt[nVar];
402  qp_obj.getPrimalSolution(xOpt);
403  isSolved = qp_obj.isSolved();
404  if (isSolved)
405  cout<<"[QP solver] Success! "<<endl;
406  else
407 
408  cout<<"[QP solver] Failure. "<<endl;
409  return Map<Matrix<double, -1, 1>>(xOpt, nVar, 1).cast<T>();
410  }

739  template<typename T,size_t dim> QpForm<T,dim> PolyTrajGen<T,dim>::getQPSet() {
740 
741  uint nVar = (this->M)*(N+1);
742  uint nineq = this->getTotalNineq(); // total line of inequality
743  uint neq = this->getTotalNeq();
744  QpForm<T,dim> qpForm(nVar,neq,nineq); // initialize all the container
745 
746  // 1. Objective function
747  spMatrixRow<T> Q(nVar,nVar);
748  for (uint d = 1; d <= this->weight_mask.size(); d++) { // dth order derivative
749  spMatrixRow<T> Qd(nVar, nVar);
750  for (uint m = 0; m < this->M; m++) { // per segment, plug in the Q
751  T dt = this->ts[m + 1] - this->ts[m];
752  sparseBlockCopy<T>(&Qd,MatrixRow<T>(IntDerSquard(d)/pow(dt,2*d-1)),m*(N+1),m*(N+1));
753  }
754  Q += this->weight_mask[d - 1] * Qd;
755  }
756 
757  for (uint dd = 0 ; dd < dim ; dd++)
758  qpForm.qpBlock[dd].Q = Q;
759 
760  uint Idx1 = 0,Idx2 = 0;
761  for (uint m = 0 ; m < this->M ; m++){ // per segment
762  // 2. Ineq constraint for this segment
763  uint Nineq = this->loosePinSet[m].size();
764  ConstraintMatPair<T,dim> Ab = loosePinMatSet((this->loosePinSet[m]).data(),this->loosePinSet[m].size());
765  for (uint dd = 0 ; dd < dim ; dd++) { // per dimension
766  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].A),(Ab.ASet[dd]),Idx1,0);
767  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].b),(Ab.bSet[dd]),Idx1,0);
768  }
769  Idx1+= 2*Nineq;
770 
771  // 3. Eq constraint for this segment
772  ConstraintMatPair<T,dim> AbFix = fixPinMatSet((this->fixPinSet[m]).data(),this->fixPinSet[m].size());
773  ConstraintMatPair<T,dim> AbConti = contiMat(m,seg_state_set[m].Nc-1);
774  uint Neq = seg_state_set[m].getN(), Nf = seg_state_set[m].Nf, Nc = seg_state_set[m].Nc;
775  for (uint dd = 0 ; dd < dim ; dd++) { // inserting A,b
776  // (Aeq)m = ([Afix ; Aconti])m : Aeq*p = beq
777  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].Aeq),(AbFix.ASet[dd]),Idx2,0) ;
778  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].Aeq),(AbConti.ASet[dd]),Idx2+Nf,0);
779  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].beq),(AbFix.bSet[dd]),Idx2,0) ;
780  sparseBlockCopy<T>(&(qpForm.qpBlock[dd].beq),(AbConti.bSet[dd]),Idx2+Nf,0);
781 
782  }
783  Idx2 += Neq;
784  }
785  return qpForm;
786  }
